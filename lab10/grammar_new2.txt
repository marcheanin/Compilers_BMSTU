'axiom PROGRAM -> RULE PROGRAM 'or 'end
RULE -> LEFT "->" RIGHT "'end" 'end

LEFT -> "'axiom" NONTERM
            'or NONTERM 'end
RIGHT -> PART_RIGHT ALTERNATIVE 'end
ALTERNATIVE -> "'or" PART_RIGHT ALTERNATIVE
                'or 'end
PART_RIGHT -> SYMB PART_RIGHT2
                'or 'end
PART_RIGHT2 -> SYMB PART_RIGHT2
                'or 'end
SYMB -> STR_TERM
            'or NONTERM 'end
